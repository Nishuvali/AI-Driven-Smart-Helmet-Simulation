import os
import numpy as np
import sounddevice as sd
import soundfile as sf
import deepspeech
import tensorflow as tf
import librosa
import librosa.display
import matplotlib.pyplot as plt
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, Dropout, BatchNormalization, Conv1D, LSTM, Flatten
import datetime
from tensorflow.keras.utils import plot_model
import pydot
import graphviz

# Ensure necessary directories exist
DATASET_PATH = "dataset/"
if not os.path.exists(DATASET_PATH):
    os.makedirs(DATASET_PATH)

# Load DeepSpeech Model
MODEL_PATH = "C:/Users/saini/deepspeech-0.9.3-models.pbmm"
SCORER_PATH = "C:/Users/saini/deepspeech-0.9.3-models.scorer"

ds_model = deepspeech.Model(MODEL_PATH)
ds_model.enableExternalScorer(SCORER_PATH)

def record_audio(duration=5, sample_rate=16000):
    """Records audio and saves it with a timestamp."""
    timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
    filename = os.path.join(DATASET_PATH, f"recording_{timestamp}.wav")
    
    print(f" Recording {duration} seconds...")
    audio_data = sd.rec(int(duration * sample_rate), samplerate=sample_rate, channels=1, dtype="int16")
    sd.wait()
    sf.write(filename, audio_data, sample_rate)
    print(f" Saved: {filename}")
    return filename

def plot_waveform(audio_path, title="Waveform"):
    """Plots the waveform of an audio file."""
    audio, sr = librosa.load(audio_path, sr=16000)
    plt.figure(figsize=(10, 4))
    librosa.display.waveshow(audio, sr=sr)
    plt.title(title)
    plt.xlabel("Time (s)")
    plt.ylabel("Amplitude")
    plt.show()

def extract_deepspeech_embedding(audio_path):
    """Extracts DeepSpeech embeddings from audio."""
    with sf.SoundFile(audio_path) as audio_file:
        audio = np.frombuffer(audio_file.read(), dtype=np.int16)
    
    # Get DeepSpeech output layer embeddings
    embeddings = ds_model.sttWithMetadata(audio, num_results=1)
    embedding_vector = np.array([ord(char.text) for char in embeddings.transcripts[0].tokens])
    
    print("Shape of extracted DeepSpeech embedding:", embedding_vector.shape)  # Debugging
    return embedding_vector

# Main Process - Record Audio First
print("Recording a new sample...")
audio_file = record_audio()

# Plot recorded audio waveform
plot_waveform(audio_file, title="Recorded Audio Waveform")

print("Extracting embeddings...")
sample_embedding = extract_deepspeech_embedding(audio_file)
input_shape_size = sample_embedding.shape[0] if sample_embedding.ndim > 0 else 1

# CNN + LSTM Model for Speech Classification
model = Sequential([
    Conv1D(filters=64, kernel_size=3, activation='relu', input_shape=(input_shape_size, 1)),
    BatchNormalization(),
    Dropout(0.3),
    LSTM(64, return_sequences=True),
    BatchNormalization(),
    Dropout(0.3),
    LSTM(32),
    Dense(128, activation='relu'),
    BatchNormalization(),
    Dropout(0.3),
    Dense(1, activation='sigmoid')
])

model.compile(loss='binary_crossentropy', optimizer='adam', metrics=['accuracy'])

def predict_speech(audio_path):
    """Predicts whether speech is intoxicated or sober."""
    embedding = extract_deepspeech_embedding(audio_path)
    embedding = np.atleast_2d(embedding)  # Ensure it's 2D
    embedding = np.expand_dims(embedding, axis=-1)  # Add channel dimension for CNN
    prediction = model.predict(embedding)[0][0]
    return " Intoxicated Speech Detected!" if prediction > 0.5 else " Sober Speech Detected."

# Predict Based on the Recorded Audio
print("Predicting...")
result = predict_speech(audio_file)
print("Prediction:", result)

# Store accuracy-related information
sample_embedding = np.atleast_2d(sample_embedding)  # Ensure it's 2D
sample_embedding = np.expand_dims(sample_embedding, axis=-1)  # Add channel dimension
history = model.fit(sample_embedding, np.array([1]), epochs=10, verbose=1)

# Plot training accuracy

plt.figure(figsize=(8, 5))
plt.plot(history.history['accuracy'], label='Accuracy')
plt.title('Model Accuracy')
plt.xlabel('Epochs')
plt.ylabel('Accuracy')
plt.legend()
plt.show()

# Visualizing the CNN + LSTM Model Architecture
try:
    plot_model(model, to_file="model_architecture.png", show_shapes=True, show_layer_names=True, dpi=300)
    plt.figure(figsize=(15, 10))
    img = plt.imread("model_architecture.png")
    plt.imshow(img)
    plt.axis('off')
    plt.title("CNN + LSTM Model Architecture")
    plt.show()
except Exception as e:
    print(" Warning: Could not generate model architecture visualization. Ensure Graphviz is installed correctly.")
    print("Error:", str(e))
